## Install

::: code-tabs#shell

@tab pnpm

```bash
pnpm add -D vuepress-plugin-comment2@next
```

@tab yarn

```bash
yarn add -D vuepress-plugin-comment2@next
```

@tab npm

```bash
npm i -D vuepress-plugin-comment2@next
```

:::

### Usage

::: code-tabs#language

@tab TS

```ts
// .vuepress/config.ts
import { commentPlugin } from "vuepress-plugin-comment2";

export default {
  plugins: [
    commentPlugin({
      // your options
    }),
  ],
};
```

@tab JS

```js
// .vuepress/config.js
import { commentPlugin } from "vuepress-plugin-comment2";

export default {
  plugins: [
    commentPlugin({
      // your options
    }),
  ],
};
```

:::
