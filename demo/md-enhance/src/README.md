### Install

```bash
pnpm add -D vuepress-plugin-md-enhance@next
```

### Usage

```ts
// .vuepress/config.ts
import { mdEnhancePlugin } from "vuepress-plugin-md-enhance";

export default {
  plugins: [
    mdEnhancePlugin({
      // your options
    }),
  ],
};
```
