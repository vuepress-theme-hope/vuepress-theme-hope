## How to use

### Install

```bash
pnpm add -D vuepress-plugin-blog2@next
```

### Usage

```ts
// .vuepress/config.ts
import { blogPlugin } from "vuepress-plugin-blog2";

export default {
  plugins: [
    blogPlugin({
      // your options
    }),
  ],
};
```
