## How to use

### Install

```bash
pnpm add -D vuepress-plugin-sitemap2
```

### Usage

```ts
// .vuepress/config.ts
import { sitemapPlugin } from "vuepress-plugin-sitemap2";

export default {
  plugins: [
    sitemapPlugin({
      // your options
    }),
  ],
};
```
