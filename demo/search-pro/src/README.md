## How to use

### Install

```bash
pnpm add -D vuepress-plugin-search-pro@next
```

### Usage

```ts
// .vuepress/config.ts
import { searchProPlugin } from "vuepress-plugin-search-pro";

export default {
  plugins: [
    searchProPlugin({
      // your options
    }),
  ],
};
```
