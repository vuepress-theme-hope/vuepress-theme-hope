## How to use

### Install

```bash
pnpm add -D vuepress-plugin-pwa2@next
```

### Usage

```ts
// .vuepress/config.ts
import { pwaPlugin } from "vuepress-plugin-pwa2";

export default {
  plugins: [
    pwaPlugin({
      // your options
    }),
  ],
};
```
